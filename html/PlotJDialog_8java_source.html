361  }

369  ColumnInfo.ColumnExtraInfo cei = rowsExtra.get(ci);
370  switch (column) {
371  case 1:
372  cei.plot = (Boolean) value;

374  break;
375  case 2:
376  cei.color = ColorEditor.colorToString((Color) value);
377  break;
378  case 3:
379  cei.linestyle = (String) value;

381  case 4:
382  cei.marker = (String) value;
